1function [ vBars ] = virtualBarsMEX( data,inc )
 2%VIRTUALBARSMEX Transforms a dataseries from a lower timeframe to higher (e.g. 1min data to 4min, 5min etc.)
 3%   This function allows the transformation of lower period observations in time to higher.  In its
 4%   current form we are only virtualizing Open and Close.  Further coding to handle High and Low should
 5%   be considered. We are accepting either a dataset or a double array as input.
 6%   Vars:   'data' is either a 2 or 4 double array which assumes standard column order of O|C or O | H | L | C
 7%           'inc' is the increment to virtualize the data to.  We must already know the input increment
 8%           e.g. 1 minute observations dataset to 4 minute virtualized would be inc = 4
 9%   The virtualization logic drops all partial bars from the end of the submitted dataset so that we are
10%   left with only full virtualized observations.  There may be need to have partial observations formed
11%   which we can revisit.
12%
13% Author: Mark Tompkins
14%
15
16%% MEX code to be skipped
17coder.extrinsic('slidefun');
18
19% Preallocate variables so we can MEX
20numBars = size(data,1);
21hTemp = zeros(numBars,1); %#ok<NASGU>
22lTemp = zeros(numBars,1); %#ok<NASGU>
23
24%% Check if we've been passed a dataset or a double array
25
26numBars = length(data);
27
28if isa(data,'double')
29    % Check if we've been passed 2 or 4 columns.
30    % We can accept 2 columns and assume Open | Close
31    % or we can accept 4 columns and assume Open | High | Low | Close
32    numCols = size(data,2);
33    if numCols == 2 || numCols == 4
34        o = data(1:inc:(floor(numBars/inc)*inc)-inc+1,1);           % Open
35        c = data(inc:inc:floor(numBars/inc)*inc,numCols);           % Close
36        vBars = [o c];
37        if numCols == 4
38            % Get the Highest high and Lowest low in the given increment
39            hTemp = slidefun('max', inc, data(:,2),'forward');      % Moving window of Max's
40            h = hTemp(1:inc:end);                                   % High
41            % Trim any overhang
42            while size(h,1) > size(o,1)
43                h(end)=[];
44            end;
45            lTemp = slidefun('min', inc, data(:,3),'forward');      % Moving window of Min's
46            l = lTemp(1:inc:end);                                   % Low
47            % Trim any overhang
48            while size(l,1) > size(o,1)
49                l(end)=[];
50            end;
51            vBars = [o h l c];
52        end; %if
53    else
54        % Seems like we might have non-standard input.  Throw an error
55        error('Input needs to be in the format of ''O | C'' or ''O | H | L | C''');
56    end; %if
57else
58    error('We need either a double array.  Datasets (or unexpected data) are not supportedin MEX.');
59end; %if
60
61end
